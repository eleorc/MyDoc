(2)class design 
abstract class Template{
	public abstract int a();
	public abstract int b();
	public abstract int c();
	public void run(){
		int state = a();
		if(state>0)
			b();
		else
			c();
	}
}
class ConcretTemplate extends Template{
	public int a(){
		return 1;
	}
	public int b(){};
	public int c(){};
}
